including tax brackets, income thresholds, historical rates, and legislative
references.

<script>{`
  let parameters = [];

  async function init() {
    try {
      const basePath = window.location.pathname.includes('/openfisca-aotearoa/')
        ? '/openfisca-aotearoa/'
        : '/';
      const response = await fetch(\`\${basePath}parameters.json\`);
      if (!response.ok) {
        throw new Error(\`Parameter catalog returned \${response.status}\`);
      }

      parameters = await response.json();
      renderList(parameters);
      setupSearch();
    } catch (error) {
      console.error('Failed to load parameters:', error);
      document.getElementById('params-list').innerHTML =
        '<div class="error">Failed to load parameters catalog.</div>';
    }
  }

  function renderList(list) {
    const listContainer = document.getElementById('params-list');
    if (list.length === 0) {
      listContainer.innerHTML = '<div class="no-results">No parameters found</div>';
      return;
    }

    listContainer.innerHTML = list.map(param => \`
      <div class="param-item" data-id="\${escapeHtml(param.id)}">
        <div class="param-item-name">\${escapeHtml(param.name)}</div>
        <div class="param-item-path">\${escapeHtml(param.path)}</div>
      </div>
    \`).join('');

    listContainer.querySelectorAll('.param-item').forEach(item => {
      item.addEventListener('click', () => {
        listContainer
          .querySelectorAll('.param-item')
          .forEach(candidate => candidate.classList.remove('active'));
        item.classList.add('active');
        showDetails(item.dataset.id);
      });
    });
  }

  function setupSearch() {
    const searchInput = document.getElementById('param-search');
    searchInput.addEventListener('input', event => {
      const query = event.target.value.toLowerCase();
      const filtered = parameters.filter(param =>
        param.name.toLowerCase().includes(query) ||
        param.path.toLowerCase().includes(query) ||
        (param.description || '').toLowerCase().includes(query)
      );
      renderList(filtered);
    });
  }

  function showDetails(id) {
    const param = parameters.find(candidate => candidate.id === id);
    const detailsContainer = document.getElementById('param-details');
    if (!param) {
      return;
    }

    let valuesHtml = '';

    if (param.type === 'brackets') {
      valuesHtml = \`
        <h4>Tax Brackets & Rates</h4>
        \${param.brackets.map((bracket, index) => renderBracket(bracket, index)).join('')}
      \`;
    } else if (param.type === 'value') {
      valuesHtml = renderValueHistory(param.values);
    } else if (param.raw) {
      valuesHtml = \`
        <h4>Raw Parameter Data</h4>
        <pre class="raw-param-json">\${escapeHtml(JSON.stringify(param.raw, null, 2))}</pre>
      \`;
    }

    detailsContainer.className = 'param-details-content';
    detailsContainer.innerHTML = \`
      <div class="detail-title">\${escapeHtml(param.name)}</div>
      <div class="detail-path">\${escapeHtml(param.path)}</div>
      <div class="detail-desc">\${escapeHtml(param.description || 'No description provided.')}</div>

      \${param.reference ? \`
        <div style="margin-bottom: 1.5rem;">
          <strong>Legislation Reference:</strong>
          <a href="\${escapeHtml(param.reference)}" target="_blank" rel="noopener">View Act/Regulations</a>
        </div>
      \` : ''}

      \${valuesHtml}
    \`;
  }

  function renderBracket(bracket, index) {
    const ratesHtml = Object.entries(bracket.rate || {}).map(([date, details]) => \`
      <tr>
        <td>\${escapeHtml(date)}</td>
        <td>\${formatPercent(details)}</td>
      </tr>
    \`).join('');

    const thresholdHtml = Object.entries(bracket.threshold || {}).map(([date, details]) => \`
      <tr>
        <td>\${escapeHtml(date)}</td>
        <td>\${formatValue(details)}</td>
      </tr>
    \`).join('');

    return \`
      <div class="bracket-card">
        <div class="bracket-card-title">Bracket \${index + 1}</div>
        <div class="bracket-grid">
          <div>
            <h5>Rates</h5>
            <table class="history-table">
              <thead><tr><th>Effective Date</th><th>Rate</th></tr></thead>
              <tbody>\${ratesHtml}</tbody>
            </table>
          </div>
          <div>
            <h5>Thresholds</h5>
            <table class="history-table">
              <thead><tr><th>Effective Date</th><th>Threshold</th></tr></thead>
              <tbody>\${thresholdHtml}</tbody>
            </table>
          </div>
        </div>
      </div>
    \`;
  }

  function renderValueHistory(values) {
    const historyHtml = Object.entries(values || {}).map(([date, details]) => \`
      <tr>
        <td>\${escapeHtml(date)}</td>
        <td>\${formatValue(details)}</td>
      </tr>
    \`).join('');

    return \`
      <h4>Value History</h4>
      <table class="history-table">
        <thead>
          <tr><th>Effective Date</th><th>Value</th></tr>
        </thead>
        <tbody>\${historyHtml}</tbody>
      </table>
    \`;
  }

  function extractValue(details) {
    return details && typeof details === 'object' && 'value' in details
      ? details.value
      : details;
  }

  function formatPercent(details) {
    const value = extractValue(details);
    return typeof value === 'number' ? \`\${(value * 100).toFixed(2)}%\` : escapeHtml(String(value));
  }

  function formatValue(details) {
    const value = extractValue(details);
    return typeof value === 'number' ? value.toLocaleString() : escapeHtml(String(value));
  }

  function escapeHtml(value) {
    return String(value)
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/"/g, '&quot;')
      .replace(/'/g, '&#39;');
  }

  init();
`}</script>
