<script>{`
  let parameters = [];
  
  async function init() {
    try {
      // Direct load of generated JSON
      const res = await fetch('/openfisca-aotearoa/parameters.json');
      parameters = await res.json();
      renderList(parameters);
      setupSearch();
    } catch (e) {
      console.error("Failed to load parameters:", e);
      document.getElementById('params-list').innerHTML = '<div class="error">Failed to load parameters catalog.</div>';
    }
  }

  function renderList(list) {
    const listContainer = document.getElementById('params-list');
    if (list.length === 0) {
      listContainer.innerHTML = '<div class="no-results">No parameters found</div>';
      return;
    }
    
    listContainer.innerHTML = list.map(param => \`
      <div class="param-item" data-id="\${param.id}">
        <div class="param-item-name">\${param.name}</div>
        <div class="param-item-path">\${param.path}</div>
      </div>
    \`).join('');
    
    // Add click listeners
    listContainer.querySelectorAll('.param-item').forEach(item => {
      item.addEventListener('click', () => {
        listContainer.querySelectorAll('.param-item').forEach(i => i.classList.remove('active'));
        item.classList.add('active');
        showDetails(item.dataset.id);
      });
    });
  }

  function setupSearch() {
    const searchInput = document.getElementById('param-search');
    searchInput.addEventListener('input', (e) => {
      const query = e.target.value.toLowerCase();
      const filtered = parameters.filter(p => 
        p.name.toLowerCase().includes(query) || 
        p.path.toLowerCase().includes(query) || 
        p.description.toLowerCase().includes(query)
      );
      renderList(filtered);
    });
  }

  function showDetails(id) {
    const param = parameters.find(p => p.id === id);
    const detailsContainer = document.getElementById('param-details');
    if (!param) return;
    
    let valuesHtml = '';
    
    if (param.type === 'brackets') {
      valuesHtml = \`
        <h4>Tax Brackets & Rates</h4>
        \${param.brackets.map((b, idx) => {
          const ratesHtml = Object.entries(b.rate).map(([date, details]) => \`
            <tr>
              <td>\${date}</td>
              <td>\${(details.value * 100).toFixed(2)}%</td>
            </tr>
          \`).join('');
          
          const threshHtml = Object.entries(b.threshold).map(([date, details]) => \`
            <tr>
              <td>\${date}</td>
              <td>$\${details.value.toLocaleString()}</td>
            </tr>
          \`).join('');
          
          return \`
            <div class="bracket-card">
              <div class="bracket-card-title">Bracket \${idx + 1}</div>
              <div style="display: grid; grid-template-columns: 1fr 1fr; gap: 1rem;">
                <div>
                  <h5>Rates</h5>
                  <table class="history-table">
                    <thead><tr><th>Effective Date</th><th>Rate</th></tr></thead>
                    <tbody>\${ratesHtml}</tbody>
                  </table>
                </div>
                <div>
                  <h5>Thresholds</h5>
                  <table class="history-table">
                    <thead><tr><th>Effective Date</th><th>Threshold</th></tr></thead>
                    <tbody>\${threshHtml}</tbody>
                  </table>
                </div>
              </div>
            </div>
          \`;
        }).join('')}
      \`;
    } else if (param.type === 'value') {
      const historyHtml = Object.entries(param.values).map(([date, details]) => \`
        <tr>
          <td>\${date}</td>
          <td>\${typeof details.value === 'number' ? '$' + details.value.toLocaleString() : details.value}</td>
        </tr>
      \`).join('');
      
      valuesHtml = \`
        <h4>Value History</h4>
        <table class="history-table">
          <thead>
            <tr><th>Effective Date</th><th>Value</th></tr>
          </thead>
          <tbody>
            \${historyHtml}
          </tbody>
        </table>
      \`;
    }
    
    detailsContainer.className = 'param-details-content';
    detailsContainer.innerHTML = \`
      <div class="detail-title">\${param.name}</div>
      <div class="detail-path">\${param.path}</div>
      <div class="detail-desc">\${param.description || 'No description provided.'}</div>
      
      \${param.reference ? \`
        <div style="margin-bottom: 1.5rem;">
          <strong>Legislation Reference:</strong> 
          <a href="\${param.reference}" target="_blank" class="detail-ref-link">View Act/Regulations ↗</a>
        </div>
      \` : ''}
      
      \${valuesHtml}
    \`;
  }

  // Kick off initialization
  init();
`}</script>
